SET SERVEROUTPUT ON ;


DECLARE
    v_customer NUMBER ;
    v_spent NUMBER ;
    
BEGIN
    SELECT customer_id, SUM(total_amount) AS spent
    INTO v_customer, v_spent
    FROM sales 
    GROUP BY customer_id
    ORDER BY 2 DESC 
    FETCH FIRST ROW ONLY
    ;
    
    DBMS_OUTPUT.PUT_LINE (v_customer) ;
END ;
